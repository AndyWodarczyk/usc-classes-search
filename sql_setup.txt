create database classes;

use classes;

DROP TABLE CLASS_DAY;
DROP TABLE CLASS_SEMESTER;
DROP TABLE CLASS_NAME;
DROP TABLE CLASS_SUBJECT;
DROP TABLE CLASS_TERM;

CREATE TABLE CLASS_TERM
(
 TERM_NAME CHAR(12),
 TERM_SORT INTEGER(2),
 PRIMARY KEY (TERM_NAME)
);

CREATE TABLE CLASS_NAME
(
 SUBJECT CHAR(4),
 COURSE_NBR CHAR(6),
 TITLE VARCHAR(96),
 VAR_CREDIT_HRS INTEGER(1),
 MIN_CREDIT_HRS FLOAT,
 MAX_CREDIT_HRS FLOAT,
 PRIMARY KEY (SUBJECT, COURSE_NBR)
);

CREATE TABLE CLASS_SEMESTER
(
 SEMESTER_TERM CHAR(12),
 CRN_NBR INTEGER(6),
 SUBJECT CHAR(4),
 COURSE_NBR CHAR(6),
 SECTION_NBR CHAR(3),
 CMP CHAR(3),
 PART_OF_TERM VARCHAR(32),
 CAPACITY INTEGER(3),
 ACTUAL INTEGER(3),
 REMAINING INTEGER(3),
 PRIMARY KEY (SEMESTER_TERM, CRN_NBR),
 FOREIGN KEY (SUBJECT, COURSE_NBR) REFERENCES CLASS_NAME (SUBJECT, COURSE_NBR)
);
 
CREATE TABLE CLASS_DAY
(
 SEMESTER_TERM CHAR(12),
 SUBJECT CHAR(4),
 CMP CHAR(3),
 CRN_NBR INTEGER(6),
 DAY_ORDER INTEGER(2),
 DAY_OF_WEEK VARCHAR(7),
 TIME_START TIME,
 TIME_END TIME,
 INSTRUCTOR VARCHAR(64),
 INSTRUCTOR_FORMAT VARCHAR(64),
 CLASS_DATE_START DATE,
 CLASS_DATE_END DATE,
 LOCATION VARCHAR(16),
 CLASS_ATTRIBUTE VARCHAR(128),
 FOREIGN KEY (SEMESTER_TERM, CRN_NBR) REFERENCES CLASS_SEMESTER (SEMESTER_TERM, CRN_NBR)
);


CREATE TABLE CLASS_SUBJECT
(
 SUBJECT_VALUE CHAR(4),
 SUBJECT_TEXT  VARCHAR(32)
);


CREATE VIEW CLASS_INSTRUCTORS
AS
SELECT CONCAT(SUBSTRING(NAME, POS + 2), IF (NAME='TBA','',', '), SUBSTRING(NAME, 1, POS)) BYLASTNAME, NAME, INSTRUCTOR
FROM
( 
 SELECT LENGTH(NAME) - LOCATE(' ',REVERSE(NAME)) POS, NAME, INSTRUCTOR
 FROM
 (
  SELECT DISTINCT
  TRIM(REPLACE(INSTRUCTOR,'(P)','')) AS NAME,
  INSTRUCTOR
  FROM CLASS_DAY
 ) a
) b;

DROP VIEW CLASS_SEARCH;

CREATE VIEW CLASS_SEARCH
AS
SELECT 
  a.SEMESTER_TERM,
  c.TITLE,
  a.CRN_NBR,
  a.SUBJECT,
  a.COURSE_NBR,
  a.SECTION_NBR,
  a.CMP,
  a.PART_OF_TERM,
  IF (c.VAR_CREDIT_HRS = 1, CONCAT(c.MIN_CREDIT_HRS,'-', c.MAX_CREDIT_HRS), c.MIN_CREDIT_HRS) CREDIT_HRS,
  a.CAPACITY,
  a.ACTUAL,
  a.REMAINING,
  b.DAY_OF_WEEK, 
  TIME_FORMAT(b.TIME_START, '%r') AS TIME_START,
  TIME_FORMAT(b.TIME_END, '%r') AS TIME_END, 
  b.INSTRUCTOR,
  b.INSTRUCTOR_FORMAT,
  DATE_FORMAT(b.CLASS_DATE_START,'%m/%d/%y') AS DATE_START_FORMATTED,
  DATE_FORMAT(b.CLASS_DATE_END,'%m/%d/%y') AS DATE_END_FORMATTED,
  b.CLASS_DATE_START AS DATE_START,
  b.CLASS_DATE_END AS DATE_END,
  b.LOCATION,
  b.DAY_ORDER
FROM
  CLASS_SEMESTER a,
  CLASS_DAY b,
  CLASS_NAME c
WHERE 
  a.SEMESTER_TERM = b.SEMESTER_TERM AND
  a.CRN_NBR = b.CRN_NBR AND 
  a.SUBJECT = c.SUBJECT AND
  a.COURSE_NBR = c.COURSE_NBR;
